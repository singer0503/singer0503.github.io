    # 调用方式

     <?php
$url = 'https://mobile.ending.best/api/business/merchant/login';
$data = json_encode(
    array(
        'token' => '*********************************** 请更换为商户 token', 
        'player_account' => 'maxtestplayer1',
        'deposit' => 100)
     );

$options = array(
    'http' => array(
        'header'  => "Content-type: application/json\r\n",
        'method'  => 'POST',
        'content' => $data,
    ),
);
$context  = stream_context_create($options);
$result = file_get_contents($url, false, $context);

var_dump(json_decode($result, true));

?>
